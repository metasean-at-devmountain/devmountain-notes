  • my original code

console.log('server running....');

var messages = [{hello: "Hello Sean"}];

var onRequest = function (req, res) {

  if (req.method == 'POST') {
    console.log('POST request ....');
    var postData = '';
    req.on('data', function(chunk) {
      console.log('on request ....');
      postData += chunk.toString();
    });    
    req.on('end', function() {
      console.log('end request ....');
      res.writeHead(201, {
        'Connection': 'close',
        'Content-Type': 'application/json',
        'Access-Control-Allow-Origin': '*'
      });
      console.log("here is the POST data:");
      console.log(JSON.parse(postData));
      messages.push(JSON.parse(postData));
      res.end(JSON.stringify(messages));
    });
  } else if (req.method == 'GET') {
      res.writeHead(200, {
      'Connection': 'close',
      'Content-Type': 'application/json',
      'Access-Control-Allow-Origin': '*'
    });
    res.end(JSON.stringify(messages));
  }
};

http = require('http');
http.createServer(onRequest).listen(3000);
  • will need ‘body-parser’, so install it

  • our collaborative (instructor-led) solution

var express = require('express')
var bodyParser = require('body-parser')
var app = express()

console.log('server running....');

app.use(bodyParser.json()) // use middleware
// https://www.npmjs.org/package/body-parser  (details lower on the page)
// http://stackoverflow.com/questions/24330014/bodyparser-is-deprecated-express-4

var messages = [{msg: "Yo, yo, yo!",
            timestamp: new Date()}];


// GET  - list all the messages
app.get('/', function(req, res) {
  console.log('GET request ....');
  res.status(333).json(messages); //status = optional
})


// POST  - create a new message
app.post('/', function(req, res) {
  console.log('POST request ....');
  var newMessage = req.body
  console.log(req.body)
  messages.push(newMessage)
  res.status(201).json(newMessage)
})

app.listen(3000)

  • additional tips

    • npm install node-dev -D ‘-D’ is like ‘–save’ but for developer
    • in an app’s package.json file you can change the scripts–>start object to node_modules/node-dev/bin/node-dev server.js (if you don’t have a start object add "start": "node_modules/node-dev/bin/node-dev server.js" as a script within the scripts, then instead of starting the server with node filename.ext run npm filenmame.ext and the server will continue to restart
    • the previous has to be done on a per-app basis; Paul showed a few of us nodemon, npm install -g nodemon then instead of node filename.ext do nodemon filename.ext to keep running the server; this only needs be done once
